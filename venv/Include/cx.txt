DECLARE
PROCEDURE UPDATE_ORG_ST
  (NIPRNAIC ORG.NAIC_ID%TYPE,
  NIPROLDLIC ORG_ST.ST_CO_NBR%TYPE,
  NEWLIC ORG_ST.ST_LIC_NBR%TYPE)
  is
  BEGIN
FOR NEXT_ORG IN(SELECT ORG_ID FROM ORG WHERE ORG.NAIC_ID = NIPRNAIC)
	LOOP
      UPDATE ORG_ST SET ST_CO_NBR = NEWLIC, ST_LIC_NBR = NEWLIC 
      WHERE ORG_ID = NEXT_ORG.ORG_ID AND ST_CD='AL' AND ST_CO_NBR = NIPROLDLIC;
    END LOOP;
    COMMIT;
  END UPDATE_ORG_ST;
 
BEGIN
 
spkg_glob.sp_InitGlobalVar('SHARK-####');
